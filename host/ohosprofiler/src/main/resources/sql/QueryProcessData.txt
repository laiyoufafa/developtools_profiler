select ta.*,ts-tb.start_ts as startTime,tc.tid,tc.pid,tc.process,tc.thread
from thread_state ta,trace_bounds tb
left join (
    select it.id,tid,pid,ip.name as process,it.name as thread from thread as it left join process ip on it.upid = ip.id
    ) tc on ta.utid = tc.id
where tc.pid = %d
and ta.cpu is not null
order by startTime
